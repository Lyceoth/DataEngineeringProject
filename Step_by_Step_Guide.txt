1. Open Virtual Machine
2. Change User to bigdata (pw: bigdata)
3. Create Folder in home folder
	name: soccerapp
	Download the files from git project as zip, unpack and save move to folder
3. open mate terminal
4. Insert to terminal preparation kubernetes
	cd soccerapp							# change directory to folder "soccerapp"
	sudo bash 
	minikube delete/minikube stop					#to go to change to root
	sudo minikube start --driver=none --memory 4096 --cpus 4
	kubectl version
	kubectl get nodes -o wide
	eval $(minikube docker-env) 					# if possible not needed
5. Insert to terminal for microservice orchestration
	minikube addons enable ingress
	eval $(minikube docker-env) # if possible 			# if possible not needed
	docker build -t my-soccer-web-app .
	kubectl apply -f 01-mysql-deployment.yaml
	kubectl apply -f 02-k8s-mysql-service.yaml
	kubectl apply -f 03-memcached-deployment-and-service.yaml
	kubectl apply -f 04-app-deployment-service-ingress.yaml
6. Use app and change details
	# Check it's running 
	watch kubectl get all
	watch kubectl get ingress -o wide
	curl http://`minikube ip`/person/l.mlb.com-p.7491

	# Scale memcached
	kubectl scale deployment memcache-deployment --replicas=2 ; watch kubectl get all
	curl http://`minikube ip`/person/l.mlb.com-p.7491

	# Scale the app
	kubectl scale deployment my-super-app-deployment --replicas=2 ; watch kubectl get all
	curl http://`minikube ip`/person/l.mlb.com-p.7491

	# Delete the app
	kubectl delete deployment/memcache-deployment
	kubectl delete deployment/my-super-app-deployment
	kubectl delete deployment/mysql-deployment
	kubectl delete svc/my-super-app-service
	kubectl delete svc/my-memcached-service
	kubectl delete svc/my-app-mysql-service
	kubectl delete ingress/my-super-app-ingress
	watch kubectl get all
